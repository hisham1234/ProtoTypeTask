-- Database: Pharmacy

-- DROP DATABASE IF EXISTS "Pharmacy";

CREATE DATABASE "Pharmacy"
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'English_United States.1252'
    LC_CTYPE = 'English_United States.1252'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;


    -- Table: public.usermaster

-- DROP TABLE IF EXISTS public.usermaster;

CREATE TABLE IF NOT EXISTS public.usermaster
(
    "薬局コード" character varying COLLATE pg_catalog."default" NOT NULL,
    "ユーザーid" character varying COLLATE pg_catalog."default" NOT NULL,
    "氏名" character varying COLLATE pg_catalog."default",
    "役職" character varying COLLATE pg_catalog."default",
    "管理者フラグ" boolean,
    "削除フラグ" boolean,
    CONSTRAINT usermaster_pkey PRIMARY KEY ("薬局コード", "ユーザーid")
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.usermaster
    OWNER to postgres;
